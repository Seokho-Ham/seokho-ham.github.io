    var addTwoNumbers = function (l1, l2) {
  //(1)
  let result = new ListNode();
  let cur = result;
  let num1 = 0;

  //(2)
  while (l1 || l2 || num1) {
    //(3)
    let sum = Math.floor((l1 ? l1.val : 0) + (l2 ? l2.val : 0) + num1);
    cur.val = sum >= 10 ? sum % 10 : sum;
    num1 = sum >= 10 ? sum / 10 : 0;
    //(4)
    if ((l1 && l1.next) || (l2 && l2.next) || num1) cur.next = new ListNode();
    cur = cur.next;
    if (l1) l1 = l1.next;
    if (l2) l2 = l2.next;
  }
  return result;
};
