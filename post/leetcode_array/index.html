    var intersect = function (nums1, nums2) {
  let result = [];
  let n1 = nums1.slice();
  let n2 = nums2.slice();
  if (nums1.length > nums2.length) {
    n1 = nums2.slice();
    n2 = nums1.slice();
  }

  for (let i = 0; i < n1.length; i++) {
    if (n2.includes(n1[i])) {
      result.push(n1[i]);
      delete n2[n2.indexOf(n1[i])];
    }
  }
  return result;
};

  if (matrix.length === 1) {
    return matrix;
  }
  let result = [];
  for (let i = 0; i < matrix.length; i++) {
    let arr = [];
    for (let j = matrix.length - 1; j >= 0; j--) {
      arr.push(matrix[j][i]);
    }
    result.push(arr);
  }
  for (let i = 0; i < result.length; i++) {
    matrix[i] = result[i];
  }
};
