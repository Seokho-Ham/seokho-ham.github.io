    let fib = [0];
let memoization = n => {
  if (n < 2) {
    fib[n] = 1;
  }

  if (!fib[n]) {
    fib[n] = memoization(n - 1) + memoization(n - 2);
  }

  return fib(n);
};

    function fib(n) {
  let arr = new Array(n);
  for (let i = 0; i < n; i++) {
    if (i === 0 || i === 1) {
      arr[i] = 1;
    } else {
      arr[i] = arr[i - 1] + arr[i - 2];
    }
  }
  return arr[n];
}
